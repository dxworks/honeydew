using System;                                                                         
namespace TopLevel
{
    public class Bar
    {
        public string Convert(int a)
        {
            return a.ToString();
        }

        public static string Cut(string s)
        {
            return s.Trim();
        }
    }

    public class Foo
    {
        private Bar _bar {get;set;}
        private string _value;

        public event Func<string> Value
        {
            add
            {
                var temp = _value;
                _bar.Convert(temp.Length);
            }
            remove => _value = Bar.Cut(value);
        }
    }
}
